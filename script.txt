// Loadout 1: ranged, e.g. double crossbow
// Loadout 4: 2x single-handed melee close quarters
//            e.g. dual-wielding swords

// Rocky plateau
?loc = rocky
  equip shovel
  ?pickup = stone
    equip star  // quick picking up
  // TODO: boss

// Deadwood Canyon
?loc = valley
  equipR hatchet
  equipL ouroboros  // healing
  ?pickup = wood
    equipR star  // quick picking up
    equipL ouroboros  // healing
  ?foe = mosquito
    loadout 1
  ?foe = scarab
    loadout 4
  ?foe.distance < 10
    loadout 4
  ?foe = boss
    loadout 4

// Caves of Fear
?loc = caves
  equipL ouroboros
  ?pickup = tar
    equipR star  // quick picking up
    equipL ouroboros  // healing
  ?foe = bat
    loadout 1
  ?foe = spider
    loadout 1
  ?foe = eggs
    loadout 1
  ?foe.distance < 10
    loadout 4
  ?foe = boss
    loadout 4

// Mushroom forest
?loc = fungus
  equip poison staff

// General rules
// Apply health potion on low health
?hp < 5
  activate potion

// Equip Experience stone to kill a boss
// and get double XP from the kill
?foe = boss & foe.armor = 0 & foe.hp < 25
  equipR sword
  equipL experience
