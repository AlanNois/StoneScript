func star_ouroboros()
  equipL star
  equipR ouroboros

func sword_shield()
  equipL sword *10
  equipR shield *10

func crossbow_shield()
  equipL crossbow *10
  equipR shield *10

func crossbow_ouroboros()
  equipL crossbow *10
  equipR ouroboros

func crossbow_speed()
  equipL crossbow *10
  equipR triskelion

func shield_ouroboros()
  equipL ouroboros
  equipR shield *10

func warhammers()
  equipL warhammer *4
  equipR warhammer *4 poison

func sword_xpstone()
  equipR sword *10
  equipL experience

func dashing()
  equipL sword *10
  equipR dash *8

func poison_staff()
  equip staff poison *10

func megasword()
  equip blade_of_god

func dps()
  equip blade_of_god
  
func armor_piercing()
  equipR warhammer *8
  equipL sword *10

func vigor_staff()
  equip staff vigor *8

func speed_shield()
  equipL triskelion
  equipR shield *10

func aether_sword_shield()
  equipL sword *7 aether
  equipR shield *10

func shovel()
  equip shovel
  
func fire_staff()
  equip staff fire *7

func healwalk()
  // Heal and gain armor while just walking
  // towards next objective. If 100% healed,
  // walk faster.
  ?target = waypoint | foe.distance > 20
    ?hp < maxhp
      shield_ouroboros()
    :
      speed_shield()
