var loadouts = import my/loadouts

// Apply health potion on low health.
?hp < 8
  activate potion

?loc = fungus_forest
  loadouts.poison_staff()
  loadouts.healwalk()


?loc = fungus_forest_boss
  // Boss, all of them
  ?foe.armor > 0
    ?foe.distance > 5
      loadouts.dashing()
    :
      loadouts.armor_piercing()
  :
    loadouts.poison_staff()
  loadouts.healwalk()
